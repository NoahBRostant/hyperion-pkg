local function writeTerminalChar(st, ch, chars)\
    ensurePartial(st)\
    local pos = st.partialCursor\
    while #st.partialCells < pos - 1 do\
        st.partialCells[#st.partialCells + 1] = makeCell(\" \", st.fg, st.bg)\
    end\
    st.partialCells[pos] = makeCell(ch, st.fg, st.bg)\
    st.partialCursor = pos + 1\
    if st.partialCursor > chars then\
        pushWrappedCells(st, st.partialCells, chars)\
        st.partialCells = {}\
        st.partialCursor = 1\
        st.partial = \"\"\
        st.partialIsPrompt = false\
        return\
    end\
    syncPartial(st)\
end\

local function parseParams(text)\
    local params = {}\
    text = tostring(text or \"\")\
    if text == \"\" then\
        return params\
    end\
    for item in (text .. \";\"):gmatch(\"([^;]*);\") do\
        params[#params + 1] = item == \"\" and 0 or (tonumber(item) or 0)\
    end\
    return params\
end\
\
local function applySgr(st, params)\
    if #params == 0 then\
        params[1] = 0\
    end\
    local i = 1\
    while i <= #params do\
        local p = params[i]\
        if p == 0 then\
            resetTerminalAttrs(st)\
        elseif p == 39 then\
            st.fg = defaultFg\
        elseif p == 49 then\
            st.bg = defaultBg\
        elseif ansiBasic[p] then\
            st.fg = ansiBasic[p]\
        elseif p >= 40 and p <= 47 then\
            st.bg = ansiBasic[p - 10] or st.bg\
        elseif p >= 100 and p <= 107 then\
            st.bg = ansiBasic[p - 10] or st.bg\
        elseif (p == 38 or p == 48) and params[i + 1] == 2 then\
            local r, g, b = params[i + 2], params[i + 3], params[i + 4]\
            local idx = nearestPalette((r or 0) * 0x10000 + (g or 0) * 0x100 + (b or 0))\
            if p == 38 then\
                st.fg = idx\
            else\
                st.bg = idx\
            end\
            i = i + 4\
        end\
        i = i + 1\
    end\
end\
\
local function handleCsi(st, paramsText, final, chars)\
    local params = parseParams(paramsText)\
    if final == \"m\" then\
        applySgr(st, params)\
    elseif final == \"K\" then\
        eraseLine(st, params[1] or 0)\
    elseif final == \"J\" then\
        if (params[1] or 0) == 2 then\
            st.lines = {}\
            st.scroll = 0\
            st.partialCells = {}\
            st.partialCursor = 1\
            st.partial = \"\"\
            st.partialIsPrompt = false\
        end\
    elseif final == \"C\" then\
        ensurePartial(st)\
        st.partialCursor = math.min(#st.partialCells + 1, st.partialCursor + math.max(1, params[1] or 1))\
    elseif final == \"D\" then\
        ensurePartial(st)\
        st.partialCursor = math.max(1, st.partialCursor - math.max(1, params[1] or 1))\
    elseif final == \"G\" then\
        ensurePartial(st)\
        st.partialCursor = math.max(1, math.min(#st.partialCells + 1, params[1] or 1))\
    elseif final == \"H\" or final == \"f\" then\
        moveTerminalCursor(st, params[1] or 1, params[2] or 1)\
    end\
    syncPartial(st)\
end\

    if syscall.type(target) ~= \"directory\" then\
        out(\"ls: not a directory: \" .. target)\
        return\
    end\

    if not ok or not list then\
        out(\"ls: cannot read: \" .. target)\
        return\
    end\
    table.sort(list)\
    if #list == 0 then\
        out(\"(empty)\")\
        return\
    end\
    local line = \"\"\
    for _, name in ipairs(list) do\
        local suffix = syscall.type(join(target, name)) == \"directory\" and \"/\" or \"\"\
        local item = name .. suffix\
        if #line + #item + 2 > 54 then\
            out(line)\
            line = item\
        elseif line == \"\" then\
            line = item\
        else\
            line = line .. \"  \" .. item\
        end\
    end\
    if line ~= \"\" then\
        out(line)\
    end\
end\

        out(\"cat: \" .. tostring(fd))\

    if not okRead then\
        out(\"cat: \" .. tostring(data))\
        return\
    end\
    data = data or \"\"\
    local count = 0\
    for line in (data .. \"\\n\"):gmatch(\"([^\\n]*)\\n\") do\
        out(line)\
        count = count + 1\
        if count >= 12 then\
            out(\"(truncated)\")\

        end\
    end\
    if #targets == 0 then\
        if not force then\
            out(\"rm: missing operand\")\
        end\
        return\
    end\
    for _, target in ipairs(targets) do\
        local ok, why = removePath(st.cwd, target, recursive, force)\
        if not ok then\
            out(\"rm: \" .. tostring(why))\
        end\
    end\
end\

local function splitPath(path)\
    local out = {}\
    for part in tostring(path or \"\"):gmatch(\"[^:]+\") do\
        out[#out + 1] = part\
    end\

local function executable(path)\
    if not path or syscall.type(path) ~= \"file\" then\
        return false\
    end\
    local ok = pcall(syscall.access, path, \"x\")\
    return ok\
end\

local function resolveExecutable(st, command)\
    if command:find(\"/\", 1, true) then\
        local path = normalize(st.cwd, command)\
        if executable(path) then\
            return path\
        end\
        return nil\
    end\
    local pathEnv = (syscall.getEnviron and syscall.getEnviron(\"PATH\")) or \"/bin/\"\
    for _, dir in ipairs(splitPath(pathEnv)) do\
        local path = dir:gsub(\"/$\", \"\") .. \"/\" .. command\
        if executable(path) then\
            return path\
        end\
    end\
    local localPath = normalize(st.cwd, command)\
    if executable(localPath) then\
        return localPath\
    end\
    return nil\
end\

local function consumeOutput(st, data, chars)\
    local changed = false\
    data = tostring(st.escapePending or \"\") .. tostring(data or \"\")\
    st.escapePending = nil\
    ensurePartial(st)\
    local i = 1\
    while i <= #data do\
        local ch = data:sub(i, i)\
        if ch == \"\\12\" then\
            st.lines = {}\
            st.partialCells = {}\
            st.partialCursor = 1\
            st.partial = \"\"\
            st.partialIsPrompt = false\
            st.scroll = 0\
            changed = true\
        elseif ch == \"\\n\" then\
            flushPartial(st, chars)\
            changed = true\
        elseif ch == \"\\r\" then\
            st.partialCursor = 1\
            syncPartial(st)\
            changed = true\
        elseif ch == \"\\b\" then\
            st.partialCursor = math.max(1, st.partialCursor - 1)\
            if st.partialCursor <= #st.partialCells then\
                table.remove(st.partialCells, st.partialCursor)\
            end\
            syncPartial(st)\
            changed = true\
        elseif ch == \"\\t\" then\
            local spaces = 4 - ((st.partialCursor - 1) % 4)\
            for _ = 1, spaces do\
                writeTerminalChar(st, \" \", chars)\
            end\
            changed = true\
        elseif ch == \"\\27\" then\
            if data:sub(i + 1, i + 1) ~= \"[\" then\
                if i == #data then\
                    st.escapePending = data:sub(i)\
                    break\
                end\
            else\
                local j = i + 2\
                while j <= #data do\
                    local code = data:byte(j)\
                    if code and code >= 0x40 and code <= 0x7E then\
                        handleCsi(st, data:sub(i + 2, j - 1), data:sub(j, j), chars)\
                        i = j\
                        changed = true\
                        break\
                    end\
                    j = j + 1\
                end\
                if j > #data then\
                    st.escapePending = data:sub(i)\
                    break\
                end\
            end\
        else\
            writeTerminalChar(st, ch, chars)\
            changed = true\
        end\
        i = i + 1\
    end\
    return changed\
end\

local function startExternal(st, args, out)\
    local executablePath = resolveExecutable(st, args[1])\
    if not executablePath then\
        out(args[1] .. \": command not found\")\
        return\
    end\
    local childArgs = {}\
    for i = 2, #args do\
        childArgs[#childArgs + 1] = args[i]\
    end\
    local opened, session = pcall(pty.open, 80, 24)\
    if not opened or not session then\
        out(args[1] .. \": pty unavailable: \" .. tostring(session))\
        return\
    end\
    local ok, pidOrErr = pcall(pty.spawn, session, executablePath, childArgs)\
    if not ok then\
        pty.close(session)\
        out(args[1] .. \": \" .. tostring(pidOrErr))\
        return\
    end\
    st.job = {\
        pty = session,\
        pid = pidOrErr,\
        command = args[1],\
    }\
end\

local function shellEnv(st)\
    local env = {}\
    for _, key in ipairs({ \"HOME\", \"USER\", \"PATH\", \"TERM\" }) do\
        local ok, value = pcall(syscall.getEnviron, key)\
        if ok and value then\
            env[key] = value\
        end\
    end\
    env.SHELL = \"/bin/gsh\"\
    env.TERM = env.TERM or \"hyperion-gfx\"\
    env.PWD = st.cwd\
    env.GSH_CWD = st.cwd\
    return env\
end\

local function startShell(st, width, height)\
    if st.job then\
        return false\
    end\
    width = tonumber(width) or 120\
    height = tonumber(height) or 80\
    local path = resolveExecutable(st, \"gsh\") or \"/bin/gsh\"\
    if syscall.type(path) ~= \"file\" then\
        pushWrapped(st, \"gsh: /bin/gsh not found\", math.max(8, math.floor((width - 10) / 4)))\
        return false\
    end\
    local cols = math.max(20, math.floor((width - 10) / 4))\
    local rows = math.max(4, visibleRows(height))\
    local opened, session = pcall(pty.open, cols, rows)\
    if not opened or not session then\
        pushWrapped(st, \"gsh: pty unavailable: \" .. tostring(session), cols)\
        return false\
    end\
    local ok, pidOrErr = pcall(pty.spawn, session, path, {}, shellEnv(st))\
    if not ok then\
        pty.close(session)\
        pushWrapped(st, \"gsh: \" .. tostring(pidOrErr), cols)\
        return false\
    end\
    st.job = {\
        pty = session,\
        pid = pidOrErr,\
        command = \"gsh\",\
        shell = true,\
    }\
    st.shellStopped = false\
    st.input = \"\"\
    st.inputCursor = 1\
    st.partial = \"\"\
    st.partialCells = {}\
    st.partialCursor = 1\
    st.partialIsPrompt = false\
    resetTerminalAttrs(st)\
    return true\
end\
\
local function runCommand(st, line, chars)\
    st.scroll = 0\
    pushWrapped(st, \"$ \" .. line, chars)\
    local args = split(line)\
    if #args == 0 then\
        return\
    end\
    local cmd = commands[args[1]]\
    local function out(text)\
        pushWrapped(st, text, chars)\
    end\
    if cmd then\
        local ok, err = pcall(cmd, st, args, out)\
        if not ok then\
            out(args[1] .. \": \" .. tostring(err))\
        end\
    else\
        startExternal(st, args, out)\
    end\
end\

local function drawTerminalLine(ctx, x, y, line, maxChars)\
    maxChars = math.max(0, math.floor(tonumber(maxChars) or 0))\
    if type(line) ~= \"table\" or type(line.cells) ~= \"table\" then\
        ui.text(ctx, x, y, ui.truncate(line, maxChars), 11, { maxWidth = ctx.width - x - 5 })\
        return\
    end\
\
    local charW = ui.textWidth(\" \", { scale = 1 })\
    for i = 1, math.min(#line.cells, maxChars) do\
        local cell = line.cells[i]\
        local cx = x + (i - 1) * charW\
        if cell and cell.bg and cell.bg ~= 0 then\
            ctx.surface:rect(cx, y - 1, charW, 7, cell.bg)\
        end\
        if cell and cell.ch ~= \" \" then\
            ui.text(ctx, cx, y, cell.ch, cell.fg or 11, { maxWidth = charW })\
        end\
    end\
end\

local function clampInputCursor(st)\
    st.input = tostring(st.input or \"\")\
    st.inputCursor = math.max(1, math.min(tonumber(st.inputCursor) or (#st.input + 1), #st.input + 1))\
end\

local function insertInput(st, ch)\
    clampInputCursor(st)\
    st.input = st.input:sub(1, st.inputCursor - 1) .. ch .. st.input:sub(st.inputCursor)\
    st.inputCursor = st.inputCursor + #ch\
end\
\
local function backspaceInput(st)\
    clampInputCursor(st)\
    if st.inputCursor <= 1 then\
        return\
    end\
    st.input = st.input:sub(1, st.inputCursor - 2) .. st.input:sub(st.inputCursor)\
    st.inputCursor = st.inputCursor - 1\
end\
\
local function deleteInput(st)\
    clampInputCursor(st)\
    if st.inputCursor <= #st.input then\
        st.input = st.input:sub(1, st.inputCursor - 1) .. st.input:sub(st.inputCursor + 1)\
    end\
end\

local function setInput(st, text)\
    st.input = tostring(text or \"\")\
    st.inputCursor = #st.input + 1\
end\

local function draw(ctx)\
    local st = ctx.state\
    local s = ctx.surface\
    s:clear(0)\
    ui.raised(ctx, 0, 0, ctx.width, ctx.height, { bg = 0, light = 3, shadow = 10 })\
    ui.text(ctx, 5, 5, \"terminal\", 5, { maxWidth = ctx.width - 10 })\
    ui.text(ctx, 46, 5, ui.truncate(st.cwd, math.floor((ctx.width - 52) / 4)), 11, { maxWidth = ctx.width - 52 })\
    ui.separator(ctx, 4, 15, ctx.width - 8, { color = 3 })\
\
    local rowH = 7\
    local y0 = 20\
    local promptY = math.max(y0, ctx.height - 10)\
    local rows = visibleRows(ctx.height)\
    local scroll = clampScroll(st, rows)\
    local total = lineCount(st)\
    local last = math.max(0, total - scroll)\
    local first = math.max(1, last - rows + 1)\
    local y = y0\
    local chars = math.max(8, math.floor((ctx.width - 10) / 4))\
    for i = first, last do\
        local line = i <= #st.lines and st.lines[i] or (st.partialIsPrompt and \"\" or { cells = st.partialCells or {} })\
        drawTerminalLine(ctx, 5, y, line, chars)\
        y = y + rowH\
    end\
\
    if maxScroll(st, rows) > 0 then\
        local trackH = math.max(8, promptY - y0 - 1)\
        local thumbH = math.max(3, math.floor(trackH * math.min(1, rows / math.max(rows, total))))\
        local travel = math.max(1, trackH - thumbH)\
        local thumbY = y0 + math.floor(travel * (1 - scroll / math.max(1, maxScroll(st, rows))))\
        s:rect(ctx.width - 4, y0, 1, trackH, 2)\
        s:rect(ctx.width - 4, thumbY, 1, thumbH, 5)\
    end\
\
    ui.panel(ctx, 4, promptY - 1, ctx.width - 8, 9, { bg = 1, border = 3 })\
    local prompt = st.shellStopped and \"[shell stopped] enter/r restart\" or (\"$ \" .. st.input)\
    ui.text(ctx, 8, promptY + 1, ui.truncate(prompt, chars), 9, { maxWidth = ctx.width - 16 })\
    if not st.shellStopped and math.floor((syscall.getUptime and syscall.getUptime() or 0) / 500) % 2 == 0 then\
        local cursorText = \"$ \" .. st.input:sub(1, math.max(0, (st.inputCursor or (#st.input + 1)) - 1))\
        local cursorX = 8 + ui.textWidth(ui.truncate(cursorText, chars), { scale = 1 })\
        s:rect(math.min(ctx.width - 9, cursorX), promptY + 1, 1, 5, 9)\
    end\
end\

function terminal.new(opts)\
    opts = opts or {}\
    local home = (syscall.getEnviron and syscall.getEnviron(\"HOME\")) or \"/\"\
    return app.new({\
        title = opts.title or \"terminal\",\
        bg = 0,\
        minWidth = 88,\
        minHeight = 58,\
        state = {\
            cwd = normalize(\"/\", opts.cwd or home),\
            input = \"\",\
            inputCursor = 1,\
            history = {},\
            historyIndex = 0,\
            scroll = 0,\
            job = nil,\
            partial = \"\",\
            partialCells = {},\
            partialCursor = 1,\
            fg = defaultFg,\
            bg = defaultBg,\
            lines = {\
                lineFromText(\"Hyperion gfx terminal\", 5, 0),\
                lineFromText(\"type help\", 11, 0),\
            },\
            maxLines = 240,\
        },\
        init = function(ctx)\
            startShell(ctx.state, ctx.width, ctx.height)\
            ctx.invalidate()\
        end,\
        resize = function(ctx)\
            ctx.invalidate()\
        end,\
        draw = draw,\
        key = function(ctx, event)\
            local st = ctx.state\
